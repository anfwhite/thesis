  <div id="controls">
    <div class="player">Player 1</div>
    <button id="rollBtn">Roll Dice</button>

    <div id="question"></div>
    <input id="answer" type="number" placeholder="Your answer">
    <button id="submitBtn">Submit Answer</button>

    <div id="log"></div>
  </div>

  <!-- ✅ JavaScript MUST be inside body -->
  <script>
    const board = document.getElementById("board");
    const rollBtn = document.getElementById("rollBtn");
    const submitBtn = document.getElementById("submitBtn");
    const questionDiv = document.getElementById("question");
    const answerInput = document.getElementById("answer");
    const log = document.getElementById("log");

    const boardSize = 25;
    let playerPos = 0;
    let correctAnswer = 0;

    // Create board
    for (let i = 0; i < boardSize; i++) {
      const cell = document.createElement("div");
      cell.className = "cell";
      cell.textContent = i;
      board.appendChild(cell);
    }

    function renderPlayer() {
      const cells = document.querySelectorAll(".cell");
      cells.forEach((cell, index) => cell.textContent = index);
      cells[playerPos].textContent = "🧍";
    }

    renderPlayer();

    rollBtn.addEventListener("click", () => {
      const roll = Math.floor(Math.random() * 6) + 1;
      playerPos = Math.min(playerPos + roll, boardSize - 1);
      renderPlayer();
      log.innerHTML += `<div>Rolled a ${roll}</div>`;
      generateQuestion();
    });

    submitBtn.addEventListener("click", () => {
      if (Number(answerInput.value) === correctAnswer) {
        log.innerHTML += `<div style="color:green">Correct!</div>`;
      } else {
        log.innerHTML += `<div style="color:red">Wrong! Answer was ${correctAnswer}</div>`;
      }
      answerInput.value = "";
      questionDiv.textContent = "";
    });

    function generateQuestion() {
      const a = Math.floor(Math.random() * 10) + 1;
      const b = Math.floor(Math.random() * 10) + 1;
      const type = Math.floor(Math.random() * 3);

      if (type === 0) {
        questionDiv.textContent = `${a} + ${b} = ?`;
        correctAnswer = a + b;
      } else if (type === 1) {
        questionDiv.textContent = `${a} - ${b} = ?`;
        correctAnswer = a - b;
      } else {
        questionDiv.textContent = `${a} × ${b} = ?`;
        correctAnswer = a * b;
      }
    }
  </script>
